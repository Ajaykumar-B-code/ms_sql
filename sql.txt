Design principles

use firstdb;

CREATE TABLE employee(
	empId INTEGER PRIMARY KEY,
	empName VARCHAR(50) NOT NULL,
	empSal DECIMAL(10,2) NOT NULL,
	job VARCHAR(50),
	phone VARCHAR(50),
	deptId INTEGER
);

INSERT INTO employee VALUES(01,'Ajay',25000,'developer','7674825164',10);

SELECT * FROM employee;

SELECT empID,empName FROM employee;

UPDATE employee SET empSal=empSal+1000;
UPDATE employee SET empSal=60000 WHERE empId =2;

UPDATE employee SET empSal=610000,deptId=25
WHERE empId =2;

DELETE FROM employee
WHERE deptId=25;

DELETE FROM employee;
 SELECT * FROM employee ORDER BY empSal;
 SELECT * FROM employee ORDER BY empSal DESC;

WHERE clause:
 SELECT * FROM employee WHERE job ='developer';
 UPDATE employee SET empSal=empSal+25000 WHERE deptId=10;
 DELETE FROM employee WHERE deptId=20;

AGGREGATE FUNCTION
SELECT MIN(empSal) FROM employee;

SELECT MAX(empSal) FROM employee;

SELECT AVG(empSal) FROM employee;

SELECT COUNT(*) FROM employee;

NUMERIC FUNCTIONS

SELECT ABS(-25);

SELECT FLOOR(25.2);

SELECT CEILING(25.2);

SELECT SIGN(-25),SIGN(25),SIGN(0);

SELECT SQUARE(6),SQRT(25),PI(),COS(30);

SELECT EXP(1);

STRING FUNCTION


 SELECT empName, LEN(empName) FROM employee;

 SELECT UPPER(empName) FROM employee;

 SELECT LOWER(empName) FROM employee;

 SELECT LTRIM('   ajay kumar   ');
 SELECT RTRIM('   ajay kumar   ');

 SELECT SUBSTRING('ajay kumar',1,4);

 SELECT REPLACE('Tester','Tester','tester');

 SELECT REPLICATE ('ajay',5);

BOOLEAN OPERATION
SELECT * FROM employee WHERE job ='developer' 
AND deptId=10;

SELECT * FROM employee WHERE job ='developer' 
OR deptId=10;

SELECT * FROM employee WHERE NOT job ='developer';

SELECT * FROM employee WHERE job !='developer';

DATE AND TIME FUNCTION

 use firstdb;
  
SELECT GETDATE() AS today_date;
select SYSDATETIME()  as todaydate;
SELECT CURRENT_TIMESTAMP AS TODAY_DATE;

SELECT DATENAME(MONTH,CURRENT_TIMESTAMP) AS MONTH;
SELECT DATENAME(YEAR,CURRENT_TIMESTAMP) AS YEAR;
SELECT DATENAME(HOUR,CURRENT_TIMESTAMP) AS HOUR;

SELECT DATEDIFF(YEAR,'JANUARY 6 1995', CURRENT_TIMESTAMP) AS AGE_IN_YEARS;
SELECT DATEDIFF(MONTH,'JANUARY 6 1995', CURRENT_TIMESTAMP) AS AGE_IN_MONTH;
SELECT DATEDIFF(YEAR,'JANUARY 6 1995', 'may 3 2002') AS AGE_IN_YEARS;

SELECT DATEADD(MONTH,10,'MAY 3 2002') AS age;
